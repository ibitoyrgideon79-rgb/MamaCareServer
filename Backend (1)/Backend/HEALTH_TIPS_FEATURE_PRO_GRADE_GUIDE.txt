This guide is intentionally backend-first. The mobile/frontend app should call one API and receive the content needed to render the Health Tips screen.

Why this feature should be built this way
1. Keep the Health Tips screen data-driven, so the app does not hardcode content forever.
2. Use DTOs for API responses so database entities do not leak to clients.
3. Use a service layer so business logic is testable without hitting PostgreSQL.
4. Use repository query methods that match the existing Spring Data JPA style in this project.
5. Use small comments only where the code needs orientation.

Official docs checked before designing this:
- Spring Boot validation reference: https://docs.spring.io/spring-boot/reference/io/validation.html
- Spring Data JPA project/reference: https://spring.io/projects/spring-data-jpa

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.EnumType;
import jakarta.persistence.Enumerated;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.GenerationType;
import jakarta.persistence.Id;
import jakarta.persistence.Index;
import jakarta.persistence.Table;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
@Entity
@Table(
        name = "health_tips",
        indexes = {
                @Index(name = "idx_health_tips_active_sort", columnList = "active, sort_order"),
                @Index(name = "idx_health_tips_category", columnList = "category")
        }
)
public class HealthTip {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 40)
    private HealthTipCategory category;

    @Column(nullable = false, length = 80)
    private String title;

    @Column(nullable = false, length = 260)
    private String body;

    @Column(name = "sort_order", nullable = false)
    @Builder.Default
    private int sortOrder = 0;

    @Column(nullable = false)
    @Builder.Default
    private boolean active = true;
}

Explanation:
HealthTip is the database model for each card on the Health Tips screen. active lets admins hide content without deleting it, and sortOrder controls the display order.

package com.Mamacare.Backend.HealthTipsFeature.Services;

import com.Mamacare.Backend.HealthTipsFeature.Dtos.HealthTipAudioGuideResponse;
import com.Mamacare.Backend.HealthTipsFeature.Dtos.HealthTipCardResponse;
import com.Mamacare.Backend.HealthTipsFeature.Dtos.HealthTipHomeResponse;
import com.Mamacare.Backend.HealthTipsFeature.Entity.HealthTip;
import com.Mamacare.Backend.HealthTipsFeature.Entity.HealthTipAudioGuide;
import com.Mamacare.Backend.HealthTipsFeature.Enums.HealthTipCategory;
import com.Mamacare.Backend.HealthTipsFeature.Repo.HealthTipAudioGuideRepo;
import com.Mamacare.Backend.HealthTipsFeature.Repo.HealthTipRepo;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.Comparator;
import java.util.List;
import java.util.Locale;

@Service
@RequiredArgsConstructor
public class HealthTipsService {

    private final HealthTipRepo healthTipRepository;
    private final HealthTipAudioGuideRepo audioGuideRepository;

    @Transactional(readOnly = true)
    public HealthTipHomeResponse getHome() {
        List<HealthTipCardResponse> tips = healthTipRepository
                .findByActiveTrueAndCategoryNotOrderBySortOrderAsc(HealthTipCategory.DAILY_TIP)
                .stream()
                .map(this::toTipResponse)
                .toList();

        HealthTipCardResponse tipOfTheDay = healthTipRepository
                .findFirstByActiveTrueAndCategoryOrderBySortOrderAsc(HealthTipCategory.DAILY_TIP)
                .map(this::toTipResponse)
                .orElse(defaultTipOfTheDay());

        return HealthTipHomeResponse.builder()
                .heading("Health Tips")
                .subtitle("Small steps, healthy you, healthy baby.")
                .audioGuides(audioGuides())
                .tips(tips.isEmpty() ? defaultTips() : tips)
                .tipOfTheDay(tipOfTheDay)
                .build();
    }

    private List<HealthTipAudioGuideResponse> audioGuides() {
        List<HealthTipAudioGuideResponse> savedGuides = audioGuideRepository
                .findByActiveTrueOrderBySortOrderAsc()
                .stream()
                .map(this::toAudioResponse)
                .sorted(Comparator.comparing(HealthTipAudioGuideResponse::getLanguage))
                .toList();

        return savedGuides.isEmpty() ? defaultAudioGuides() : savedGuides;
    }

    private HealthTipCardResponse toTipResponse(HealthTip tip) {
        return HealthTipCardResponse.builder()
                .id(tip.getId())
                .category(tip.getCategory())
                .title(tip.getTitle())
                .body(tip.getBody())
                .sortOrder(tip.getSortOrder())
                .build();
    }

    private HealthTipAudioGuideResponse toAudioResponse(HealthTipAudioGuide audioGuide) {
        return HealthTipAudioGuideResponse.builder()
                .id(audioGuide.getId())
                .language(audioGuide.getLanguage())
                .title(audioGuide.getTitle())
                .audioUrl(audioGuide.getAudioUrl())
                .durationSeconds(audioGuide.getDurationSeconds())
                .displayDuration(displayDuration(audioGuide.getDurationSeconds()))
                .build();
    }

    private String displayDuration(int durationSeconds) {
        int minutes = durationSeconds / 60;
        int seconds = durationSeconds % 60;
        return String.format(Locale.ENGLISH, "%d.%02d min", minutes, seconds);
    }

    private List<HealthTipAudioGuideResponse> defaultAudioGuides() {
        return List.of(
                HealthTipAudioGuideResponse.builder()
                        .language(com.Mamacare.Backend.HealthTipsFeature.Enums.HealthTipLanguage.YORUBA)
                        .title("Listen in yoruba")
                        .audioUrl("/audio/health-tips-yoruba.mp3")
                        .durationSeconds(75)
                        .displayDuration("1.15 min")
                        .build()
        );
    }

    private List<HealthTipCardResponse> defaultTips() {
        return List.of(
                defaultTip(HealthTipCategory.NUTRITION, "Nutrition", "Eat healthy for you and your baby.", 1),
                defaultTip(HealthTipCategory.HYDRATION, "Stay Hydrated", "Drink enough water daily.", 2),
                defaultTip(HealthTipCategory.REST, "Get Enough Rest", "Good rest supports a healthy pregnancy.", 3),
                defaultTip(HealthTipCategory.STRESS_MANAGEMENT, "Manage Stress", "Stay calm and think positive.", 4)
        );
    }

    private HealthTipCardResponse defaultTipOfTheDay() {
        return defaultTip(
                HealthTipCategory.DAILY_TIP,
                "Tips of the Day",
                "Eat variety of fruits and vegetables everyday.",
                5
        );
    }

    private HealthTipCardResponse defaultTip(
            HealthTipCategory category,
            String title,
            String body,
            int sortOrder
    ) {
        return HealthTipCardResponse.builder()
                .category(category)
                .title(title)
                .body(body)
                .sortOrder(sortOrder)
                .build();
    }
}

Explanation:
The service returns saved database content when available, but falls back to the exact screen copy you provided. This means the feature works on day one and can later become admin-editable.

Important correction:
The screen says "1.25 min". If that means 1 minute 25 seconds, use durationSeconds = 85 and displayDuration = "1.25 min". If it means 75 seconds, display it as "1.15 min". Pick one format and stay consistent. I recommend "1:25" for frontend display because users understand time better that way.

package com.Mamacare.Backend.HealthTipsFeature.Services;

import com.Mamacare.Backend.HealthTipsFeature.Dtos.HealthTipHomeResponse;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import java.util.List;
import java.util.Optional;

import static org.assertj.core.api.Assertions.assertThat;
import static org.mockito.Mockito.when;

@ExtendWith(MockitoExtension.class)
class HealthTipsServiceTest {

    private HealthTipsService healthTipsService;

    @Mock
    private HealthTipRepo healthTipRepository;

    @Mock
    private HealthTipAudioGuideRepo audioGuideRepository;

    @BeforeEach
    void setUp() {
        healthTipsService = new HealthTipsService(healthTipRepository, audioGuideRepository);
    }

    @Test
    void getHomeReturnsSavedTipsAndAudioGuide() {
        HealthTip nutrition = HealthTip.builder()
                .id(1L)
                .category(HealthTipCategory.NUTRITION)
                .title("Nutrition")
                .body("Eat healthy for you and your baby.")
                .sortOrder(1)
                .active(true)
                .build();
        HealthTip dailyTip = HealthTip.builder()
                .id(2L)
                .category(HealthTipCategory.DAILY_TIP)
                .title("Tips of the Day")
                .body("Eat variety of fruits and vegetables everyday.")
                .sortOrder(5)
                .active(true)
                .build();
        HealthTipAudioGuide audioGuide = HealthTipAudioGuide.builder()
                .id(3L)
                .language(HealthTipLanguage.YORUBA)
                .title("Listen in yoruba")
                .audioUrl("/audio/health-tips-yoruba.mp3")
                .durationSeconds(85)
                .sortOrder(1)
                .active(true)
                .build();

        when(healthTipRepository.findByActiveTrueAndCategoryNotOrderBySortOrderAsc(HealthTipCategory.DAILY_TIP))
                .thenReturn(List.of(nutrition));
        when(healthTipRepository.findFirstByActiveTrueAndCategoryOrderBySortOrderAsc(HealthTipCategory.DAILY_TIP))
                .thenReturn(Optional.of(dailyTip));
        when(audioGuideRepository.findByActiveTrueOrderBySortOrderAsc())
                .thenReturn(List.of(audioGuide));

        HealthTipHomeResponse response = healthTipsService.getHome();

        assertThat(response.getHeading()).isEqualTo("Health Tips");
        assertThat(response.getSubtitle()).isEqualTo("Small steps, healthy you, healthy baby.");
        assertThat(response.getTips()).hasSize(1);
        assertThat(response.getTips().get(0).getTitle()).isEqualTo("Nutrition");
        assertThat(response.getTipOfTheDay().getTitle()).isEqualTo("Tips of the Day");
        assertThat(response.getAudioGuides().get(0).getLanguage()).isEqualTo(HealthTipLanguage.YORUBA);
        assertThat(response.getAudioGuides().get(0).getDisplayDuration()).isEqualTo("1.25 min");
    }

    @Test
    void getHomeUsesDefaultsWhenDatabaseHasNoContent() {
        when(healthTipRepository.findByActiveTrueAndCategoryNotOrderBySortOrderAsc(HealthTipCategory.DAILY_TIP))
                .thenReturn(List.of());
        when(healthTipRepository.findFirstByActiveTrueAndCategoryOrderBySortOrderAsc(HealthTipCategory.DAILY_TIP))
                .thenReturn(Optional.empty());
        when(audioGuideRepository.findByActiveTrueOrderBySortOrderAsc())
                .thenReturn(List.of());

        HealthTipHomeResponse response = healthTipsService.getHome();

        assertThat(response.getTips()).hasSize(4);
        assertThat(response.getTips().get(1).getTitle()).isEqualTo("Stay Hydrated");
        assertThat(response.getTipOfTheDay().getBody())
                .isEqualTo("Eat variety of fruits and vegetables everyday.");
        assertThat(response.getAudioGuides()).hasSize(1);
    }
}

This is a pure service unit test. It mocks repositories, so it does not need PostgreSQL and it will not fail because of database credentials.
